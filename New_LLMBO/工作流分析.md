```python
def generate_warmstart_candidates(
    self,
    n: int = 15,
    batch_size: int = 20,
    max_attempts: int = 4,
) -> List[np.ndarray]:
    all_candidates: List[np.ndarrayARRAY] = []

    # 分批调用 LLM
    for batch_idx in range(max_attempts):
        if len(all_candidates) >= n:
            break

        prompt = self._render_warmstart_prompt(batch_size)

        batch: List[np.ndarray] = []
        for retry_idx in range(max_retries + 1):
            responses = self._caller.call(
                prompt, temperature=0.7, max_tokens=2500)
            batch = self._parser.parse_candidates(responses)
            if batch:
                break

        # 去重并收集
        for cand in batch:
            h = tuple(cand.round(4).tolist())
            if h not in seen:
                all_candidates.append(cand)

    # 不足则用物理启发式补全
    if len(all_candidates) < n:
        all_candidates.extend(
            self._fallback.physics_informed_warmstart(n - len(all_candidates))
        )

    return all_candidates[:n]
```

**实现逻辑：**
- 渲染 WarmStart Prompt 模板（包含电池参数、约束）
- 分批调用 LLM（每批最多 20 个候选）
- **ResponseParser**：提取 JSON，验证边界和 dSOC 约束
- **失败回退**：用物理启发式（7 个先验点 + LHS）
- 缓存结果供后续初始化 μ 使用

```python
def physics_informed_warmstart(self, n: int) -> List[np.ndarray]:
    # 7 个基于领域知识的先验点
    prior_points = [
        np.array([5.5, 4.5, 2.8, 0.20, 0.20]),  # 激进快充
        np.array([2.5, 2.5, 2.0, 0.35, 0.25]),  # 保守安全
        np.array([4.0, 3.5, 2.5, 0.25, 0.20]),  # 均衡折衷
        np.array([5.0, 4.0, 2.2, 0.20, 0.25]),  # 偏快，温度控制
        np.array([3.5, 3.0, 2.0, 0.30, 0.28]),  # 低老化
        np.array([5.8, 3.0, 2.0, 0.18, 0.22]),  # 大 I1 启动
        np.array([3.0, 2.8, 2.2, 0.38, 0.28]),  # 平衡温度和老化
    ]

    candidates = prior_points[:min(n, 7)]

    if len(candidates) < n:
        candidates.extend(self.lhs_candidates(n - len(candidates)))  # LHS 补全

    return candidates[:n]
```

**实现逻辑：**
- 提供覆盖 Pareto 极端方向的先验点
- 激进：高电流、小 SOC 区间（快但热）
- 保守：低电流、大 SOC 区间（慢但安全）
- 不足则用 LHS 补全

```python
def run_optimization_loop() -> None:
    for t in range(int(self.cfg["max_iterations"])):
        # 1. Bootstrap：如果可行解不足，添加随机点
        if self.database.n_feasible < 2:
            theta = self._lhs_candidates(1)[0]
            result = self.simulator.evaluate(theta)
            self.database.add_from_simulator(...)
            continue

        # 2. 采样 Riesz 权重
        w_vec = self._next_weight()

        # 3. 更新动态边界和 Tchebycheff 上下文
        self._update_dynamic_bounds()
        X_train, Y_raw = self.database.get_train_XY(feasible_only=True)
        ideal_point_raw = self._compute_dynamic_ideal_point(Y_raw)
        self.database.update_tchebycheff_context(...)

        # 4. 计算 Tchebycheff 标量值
        scalar_y = compute_tchebycheff_from_raw_with_ideal(
            Y_raw=Y_raw, w_vec=w_vec,
            ideal_point_raw=ideal_point_raw,
            y_min=self._y_tilde_min, y_max=self._y_tilde_max,
            eta=0.05,
        )

        # 5. 拟合 GP
        self.gp.fit(X_train, scalar_y, w_vec=w_vec, t=t)

        # 6. LLM 迭代指导
        guidance = None
        coupling = None
        X_candidates = None

        if self.cfg["enable_iterative_guidance"]:
            uncertainty_hotspots = self._compute_uncertainty_hotspots(t)
            guidance_state = self._build_guidance_state(...)
            guidance = self.llm.query_iteration_guidance(guidance_state)

            if guidance is not None:
                coupling, guidance_candidates = self._build_gp_llm_coupling_from_guidance(
                    guidance, t
                )
                X_candidates = guidance_candidates

        # 7. 采集函数选择候选点
        acq_result = self.af.step(
            X_candidates=X_candidates,
            database=self.database,
            t=t,
            w_vec=w_vec,
            lift=coupling,  # GP-LLM 耦合
        )

        # 8. 评估选定的点
        for rank, theta in enumerate(acq_result.selected_thetas):
            sim_result = self.simulator.evaluate(theta)
            self.database.add_from_simulator(
                theta=theta, result=sim_result,
                source="bo", iteration=t + 1,
                acq_value=float(acq_result.selected_scores[rank]),
                gp_pred={...},
                llm_rationale=json.dumps(guidance.to_dict()),
            )

        # 9. 记录统计信息
        self.database.record_iteration_stats(...)

        # 10. 定期保存检查点
        if (t + 1) % int(self.cfg["checkpoint_every"]) == 0:
            self._save_checkpoint(t)
```

1. **Bootstrap**：如果可行解 < 2，添加 LHS 点

5. **GP 拟合**：5D 输入 → 1D 标量值

9. **仿真评估**：在 PyBaMM 上评估选定协议

```python
def _next_weight(self) -> np.ndarray:
    if not self._weight_order:
        # 第一次：打乱顺序
        order = self._rng.permutation(len(self._weight_set))
        self._weight_order = order.tolist()
    # 顺序弹出
    return self._weight_set[self._weight_order.pop()]
```

**实现逻辑：**

- 集合耗尽后重新打乱，确保均匀遍历
- 保证每次迭代的权重不同，覆盖 Pareto 前沿

```python
def _update_dynamic_bounds() -> None:
    feasible = self.database.get_feasible()
    if not feasible:
        self._y_tilde_min = np.zeros(3)
        self._y_tilde_max = np.ones(3)
        return

    Y_raw = np.array([obs.objectives for obs in feasible])
    Y_tilde = log_transform_objectives(Y_raw)  # log10 变换
    self._y_tilde_min, self._y_tilde_max = compute_dynamic_bounds(Y_tilde)

    # 如果范围太小，用全局范围兜底（避免归一化不稳定）
    global_range = [
        np.log10(REF_POINT[0]) - np.log10(IDEAL_POINT[0]),  # time
        REF_POINT[1] - IDEAL_POINT[1],                        # temp
        np.log10(REF_POINT[2]) - np.log10(IDEAL_POINT[2]),  # aging
    ]

    for i in range(3):
        if (self._y_tilde_max[i] - self._y_tilde_min[i]) < 0.05 * global_range[i]:
            self._y_tilde_min[i] = global_min[i]
            self._y_tilde_max[i] = global_max[i]
```

**实现逻辑：**
- 从历史可行解计算 y_min, y_max
- 对 time 和 aging 应用 log10 变换
- 如果范围过小（< 5% 全局范围），用全局范围兜底
- 避免早期迭代归一化不稳定

#### 5.4 Tchebycheff 标量化
**文件：`llmbo/optimizer.py`**
**方法：`compute_tchebycheff_from_raw_with_ideal()` (第119-132行)**

```python
def compute_tchebycheff_from_raw_with_ideal(
    Y_raw: np.ndarray,
    w_vec: np.ndarray,
    ideal_point_raw: np.ndarray,
    y_min: np.ndarray,
    y_max: np.ndarray,
    eta: float = 0.05,
) -> np.ndarray:
    # 1. Log10 变换
    Y_tilde = log_transform_objectives(Y_raw)

    # 2. 映射到理想点
    ideal_tilde = log_transform_objectives(ideal_point_raw[None, :])[0]

    # 3. 归一化（相对于全局范围）
    denom = y_max - y_min
    Y_gap = np.abs(Y_tilde - ideal_tilde[np.newaxis, :]) / denom[np.newaxis, :]

    # 4. Tchebycheff 标量化 + η tiebreaker
    weighted = Y_gap * w_vec[np.newaxis, :]
    return weighted.max(axis=1) + float(eta) * weighted.sum(axis=1)
```

**实现逻辑：**
- **Log10 变换**：对 time 和 aging 取 log，temp 保持
- **归一化**：缩放到 [0,1] 区间
- **Tchebycheff**：`max_i(w_i * f̄_i) + η * Σ(w_i * f̄_i)`
- **η = 0.05**：打破 ties 的平滑项

#### 5.5 GP 拟合
**文件：`llmbo/gp_model.py`**
**方法：`MaternGPModel.fit()` (第166-228行)**

```python
def fit(
    self,
    X: np.ndarray,
    y: np.ndarray,
    w_vec: Optional[np.ndarray] = None,
    t: int = 0,
) -> "MaternGPModel":
    # 1. 归一化输入到 [0,1]
    X_norm = self._normalize_X(X)

    # 2. 构建 Matern 核 + WhiteKernel
    kernel = (
        ConstantKernel(1.0, (1e-3, 1e3))
        * Matern(length_scale=np.ones(5), length_scale_bounds=(1e-3, 1e3), nu=2.5)
        + WhiteKernel(noise_level=1e-6, noise_level_bounds=(1e-12, 1e-1))
    )

    # 3. 创建 GP 拟合器
    self._model = GaussianProcessRegressor(
        kernel=kernel,
        alpha=self.alpha,
        normalize_y=True,
        n_restarts_optimizer=5,  # 多起点优化超参数
        random_state=self.random_state,
    )

    # 4. 拟合
    with warnings.catch_warnings():
        warnings.simplefilter("ignore", ConvergenceWarning)
        self._model.fit(X_norm, y)

    # 5. 保存训练摘要
    self._last_summary = {
        "n_train": int(X.shape[0]),
        "kernel": str(self._model.kernel_),
        "length_scale": np.asarray(self._model.kernel_.k1.k2.length_scale).tolist(),
        "noise": float(self._model.kernel_.k2.noise_level),
        "y_best": float(np.min(y)),
        "iteration": int(t),
    }
    return self
```

**实现逻辑：**
- **输入归一化**：将 5D 决策变量缩放到 [0,1]
- **核函数**：Matern(ν=2.5) + 白噪声

```python

    self,
    guidance: IterationGuidance,
    t: int,
) -> Tuple[Any, np.ndarray]:
    if guidance.mode == "region":
        # Region 模式：在 [lb, ub] 上 Sobol 采样网格
        grid = self._sobol_grid(
            guidance.lb, guidance.ub,
            n_points=64, seed=2000 + t,
        )
        weights = np.full(grid.shape[0], 1.0 / grid.shape[0])  # 均匀权重
    else:
        # Point 模式：以点为中心高斯采样
        center = guidance.representative_point()
        sigma = self._estimate_search_sigma() * 0.75
        grid = self._sobol_grid(center - 2*sigma, center + 2*sigma, 25)
        grid = np.vstack([center[None, :], grid])
        # 高斯权重：离中心越近权重越大
        diff = (grid - center) / sigma
        weights = np.exp(-0.5 * np.sum(diff ** 2, axis=1))

    # 耦合校准：λ = c / √(aᵀ Σ_GG a)
    coupling = self.gp.build_preference_coupling(
        grid=grid,
        weights=weights,
        confidence=guidance.confidence,
        mode=guidance.mode,
    )

    return coupling, grid
```

**实现逻辑：**
- **Region 模式**：在 LLM 指定的边界内均匀采样网格

- **Point 模式**：以推荐点为中心高斯采样
- **权重归一化**：`weights = weights / sum(weights)`
- **耦合校准**：在 GP 中计算 λ 和后验方差

#### 5.10 构建偏好耦合
**文件：`llmbo/gp_model.py`**
**方法：`build_preference_coupling()` (第280-311行)**

```python
def build_preference_coupling(
    self,
    grid: np.ndarray,
    weights: np.ndarray,
    confidence: float,
    mode: str = "region",
) -> LLMPreferenceCoupling:
    # 1. 归一化权重
    weights = np.clip(weights, 0.0, None)
    weight_sum = float(weights.sum())
    weights = weights / weight_sum

    # 2. 计算网格点的后验协方差
    sigma_gg = self.posterior_covariance(grid, grid)
    posterior_variance = float(weights @ sigma_gg @ weights)

    # 3. 校准 λ：λ = c / √(Var[wᵀ f])
    lambda_value = float(
        np.clip(confidence, 0.0, 1.0) /
        np.sqrt(max(posterior_variance, 1e-12))
    )

    return LLMPreferenceCoupling(
        mode=str(mode),
        grid=grid,
        weights=weights,
        confidence=float(confidence),
        lambda_value=lambda_value,
        posterior_variance=posterior_variance,
    )
```

**实现逻辑：**
- **协方差计算**：`σ_GG = Cov[f(G), f(G)]`（网格点的 GP 后验协方差）
- **后验方差**：`Var[wᵀ f] = wᵀ σ_GG w`
- **λ 校准**：确保偏移量与不确定性尺度匹配
- **λ 的物理意义**：每单位标准差的偏移强度

```python
def step(
    self,
    X_candidates: Optional[np.ndarray],
    database: DatabaseProtocol,
    t: int,
    w_vec: Optional[np.ndarray],
    lift: Optional[Any] = None,

    f_min = database.get_f_min()
    theta_best = self._repair_theta(database.get_theta_best())
    stagnation_count = database.get_stagnation_count()

    # 2. 停滞时扩大搜索范围 σ
    sigma_scale = 1.0 + 0.20 * min(stagnation_count, 3)

    # 3. 构建候选池
    candidate_pool = self._build_candidate_pool(
        theta_best, X_candidates, f_min, lift=lift
    )

    # 4. 预测（带耦合偏移）
    mean, std = self.gp.predict_with_coupling(candidate_pool, coupling=lift)
    mean_base = self.gp.predict(candidate_pool)[0] if lift is not None else mean.copy()

    # 5. 计算 Expected Improvement
    ei = expected_improvement(mean, std, f_min)
    wcharge = np.ones_like(ei)
    score = ei * wcharge

    # 6. 选择分数最高的候选点

        selected_thetas=selected_thetas,
        selected_scores=score[selected_indices],
        all_mean=mean,
        all_mean_base=mean_base,
        all_std=std,
        all_ei=ei,
        state=self.get_state(),
    )
```

- **候选池构建**：LLM 点 + 高斯采样 + 均匀采样 + L-BFGS-B 优化

- **EI 计算**：`EI = (f_min - μ)Φ(z) + σφ(z)`
- **Top-K 选择**：贪心选择 EI 最高的点

#### 5.12 GP 预测（带耦合）
**文件：`llmbo/gp_model.py`**
**方法：`predict_with_coupling()` (第240-251行)**

```python
def predict_with_coupling(
    self,
    X_new: np.ndarray,
    coupling: Optional[LLMPreferenceCoupling] = None,
) -> Tuple[np.ndarray, np.ndarray]:
    # 1. 基础预测
    mean, std = self.predict(X_new)

    if coupling is None:
        return mean, std

    # 2. 计算耦合偏移（论文 Line 12）
    # σ_θG = Cov[f(θ), f(G)]
    sigma_xg = self.posterior_covariance(X_new, coupling.grid)

    # shift = λ · aᵀ · σ_θG
    shift = float(coupling.lambda_value) * np.asarray(
        sigma_xg @ coupling.weights, dtype=float
    ).ravel()

    # 均值偏移，标准差不变
    return mean - shift, std
```

**实现逻辑：**
- **基础预测**：标准 GP 预测（μ, σ）
- **协方差计算**：`σ_θG = Cov[f(θ_new), f(G)]`（候选点与网格的协方差）
- **偏移计算**：`shift = λ · weightsᵀ · σ_θG`
- **返回**：`μ_lifted = μ - shift`，σ 保持不变

#### 5.13 Expected Improvement 计算
**文件：`llmbo/acquisition.py`**
**方法：`expected_improvement()` (第335-342行)**

```python
def expected_improvement(
    mean: np.ndarray,
    std: np.ndarray,
    f_min: float
) -> np.ndarray:
    mean = np.asarray(mean).ravel()
    std = np.clip(np.asarray(std).ravel(), 1e-12, None)

    improvement = f_min - mean  # 潜在改进量
    z = improvement / std  # 标准化改进

    # EI = (f_min - μ)Φ(z) + σφ(z)
    ei = improvement * scipy_norm.cdf(z) + std * scipy_norm.pdf(z)

    # σ → 0 时 EI → 0
    ei[std <= 1e-12] = 0.0
    return np.maximum(ei, 0.0)
```

**实现逻辑：**
- **潜在改进**：`Δf = f_min - μ`
- **标准化**：`z = Δf / σ`
- **EI 公式**：
  - `Φ(z)`：标准正态 CDF
  - `φ(z)`：标准正态 PDF
  - `EI = Δf · Φ(z) + σ · φ(z)`
- **边界处理**：σ ≤ ε 时 EI = 0

```python
def _build_candidate_pool(
    self,
    theta_best: np.ndarray,
    X_candidates: Optional[np.ndarray],
    f_min: float,
    lift: Optional[Any] = None,
) -> np.ndarray:
    pool: List[np.ndarray] = []

    # 1. LLM 提供的候选点
    if X_candidates is not None:
        pool.extend(X_candidates)

    # 2. 当前最优解
    pool.append(theta_best.copy())

    # 3. 高斯采样（围绕 μ）
    pool.extend(self._sample_gaussian(
        self.n_restarts_optimizer,
        self._state.mu,
        self._state.sigma
    ))

    # 4. 均匀采样
    pool.extend(self._sample_uniform(self.n_random_candidates))

    # 5. L-BFGS-B 局部优化 EI
    seeds = self._deduplicate(pool)
    optimized: List[np.ndarray] = []
    for seed in seeds[: self.n_restarts_optimizer]:
        optimized.append(self._optimize_from_seed(seed, f_min, lift=lift))

    pool.extend(optimized)
    pool = self._deduplicate(pool)

    return np.vstack(pool)
```

- **LLM 候选**：优先探索 LLM 建议的区域

- **高斯采样**：围绕 μ ± σ 采样
- **均匀采样**：全局探索
- **L-BFGS-B 优化**：从每个 seed 点出发优化 EI

#### 5.15 L-BFGS-B 优化 EI

```python
def _optimize_from_seed(
    self,
    seed: np.ndarray,
    f_min: float,
    lift: Optional[Any] = None,
) -> np.ndarray:
    x0 = self._repair_theta(seed)

    def objective(x: np.ndarray) -> float:
        x = self._repair_theta(x)
        mean, std = self.gp.predict_with_coupling(x[None, :], coupling=lift)
        return -float(expected_improvement(mean, std, f_min)[0])  # 最小化 -EI

    try:
        result = minimize(
            objective,
            x0=x0,
            method="L-BFGS-B",
            bounds=self._bounds,  # I1/I2/I3/dSOC1/dSOC2 的边界
            options={"maxiter": 100},
        )
        if result.success:
            return self._repair_theta(result.x)
    except Exception as exc:
        logger.debug("L-BFGS-B failed: %s", exc)

    return x0
```

**实现逻辑：**
- **目标函数**：最小化 `-EI(θ)`（等价于最大化 EI）
- **边界约束**：5 维决策变量的上下界
- **L-BFGS-B**：有界拟牛顿法，快速局部优化
- **约束修复**：每次迭代后检查 dSOC1 + dSOC2 ≤ 0.70

#### 6.3 更新 Pareto 前沿

```python
def _update_pareto(self, new_obj: Optional[np.ndarray] = None) -> None:
    feasible = [(i, o) for i, o in enumerate(self._observations) if o.feasible]
    if not feasible:
        self._pareto_indices = []
        self._pareto_objectives = None
        return

    if new_obj is None:
        # 全量重建（首次或 load 时）
        indices = [i for i, _ in feasible]
        objs = np.array([o.objectives for _, o in feasible])
        n = len(objs)
        is_dominated = np.zeros(n, dtype=bool)

        for i in range(n):
            if is_dominated[i]:
                continue
            for j in range(n):
                if i == j or is_dominated[j]:
                    continue
                # 检查 j 是否支配 i
                if np.all(objs[j] <= objs[i]) and np.any(objs[j] < objs[i]):
                    is_dominated[i] = True
                    break

        self._pareto_indices = [indices[i] for i in range(n) if not is_dominated[i]]
        self._pareto_objectives = objs[~is_dominated]
        return

    # 增量更新（每次添加可行解时）
    new_obj = np.asarray(new_obj)
    new_idx = len(self._observations) - 1

    if not self._pareto_indices:
        self._pareto_indices = [new_idx]
        self._pareto_objectives = new_obj[np.newaxis, :]
        return

    # 检查新点是否被当前 Pareto 支配
    for pf_obj in self._pareto_objectives:
        if np.all(pf_obj <= new_obj) and np.any(pf_obj < new_obj):
            return  # 被支配，不添加

    # 移除被新点支配的旧点
    not_dominated_by_new = ~(
        np.all(new_obj <= self._pareto_objectives, axis=1) &
        np.any(new_obj < self._pareto_objectives, axis=1)
    )
    self._pareto_indices = [
        self._pareto_indices[i]
        for i in range(len(self._pareto_indices)) if not_dominated_by_new[i]
    ]
    self._pareto_objectives = self._pareto_objectives[not_dominated_by_new]

    # 添加新点
    self._pareto_indices.append(new_idx)
    self._pareto_objectives = np.vstack([self._pareto_objectives, new_obj])
```

**实现逻辑：**
- **全量重建**：首次或 load 时，O(n²) 支配检查
- **增量更新**：每次添加可行解
  - 检查新点是否被支配（如果是，直接返回）
  - 移除被新点支配的旧点
  - 添加新点到前沿
- **支配定义**：`j 支配 i` 当且仅当 `∀k, f_j[k] ≤ f_i[k]` 且 `∃k, f_j[k] < f_i[k]`

```python
def _recompute_best(self, update_stagnation: bool = False) -> None:
    feasible = self.get_feasible()
    if not feasible:
        return

    Y_raw = np.array([o.objectives for o in feasible])

    # 1. Log10 变换
    Y_tilde = Y_raw.copy()
    Y_tilde[:, 0] = np.log10(np.maximum(Y_raw[:, 0], 1.0))
    Y_tilde[:, 2] = np.log10(np.maximum(Y_raw[:, 2], 1e-12))

    # 2. 动态归一化
    global_range = np.array([
        np.log10(DEFAULT_REF_POINT[0]) - np.log10(DEFAULT_IDEAL_POINT[0]),
        DEFAULT_REF_POINT[1] - DEFAULT_IDEAL_POINT[1],
        np.log10(DEFAULT_REF_POINT[2]) - np.log10(DEFAULT_IDEAL_POINT[2]),
    ])
    denom = self._y_max - self._y_min
    for i in range(3):
        if denom[i] < 0.05 * global_range[i]:
            denom[i] = global_range[i]

    if self._ideal_point_raw is not None:
        ideal_tilde = self._ideal_point_raw[np.newaxis, :].copy()
        ideal_tilde[:, 0] = np.log10(np.maximum(ideal_tilde[:, 0], 1.0))
        ideal_tilde[:, 2] = np.log10(np.maximum(ideal_tilde[:, 2], 1e-12))
        Y_bar = np.abs(Y_tilde - ideal_tilde[0]) / denom
    else:
        Y_bar = (Y_tilde - self._y_min) / denom

    # 3. Tchebycheff 标量化
    w = self._w_vec
    Wf = w[np.newaxis, :] * Y_bar
    F_tch = Wf.max(axis=1) + self._eta * Wf.sum(axis=1)

    # 4. 找最优
    best_idx = int(np.argmin(F_tch))
    self._prev_f_min = self._f_min
    self._f_min = float(F_tch[best_idx])
    self._theta_best = feasible[best_idx].theta.copy()

    # 5. 更新停滞计数
    if update_stagnation:
        current_hv = self.compute_hypervolume()
        current_pareto_size = self.pareto_size
        hv_improvement = current_hv - self._prev_hv_for_stagnation
        pf_grew = current_pareto_size > self._prev_pareto_size
        improved = (hv_improvement > 1e-3) or pf_grew

        self._improvement_window.append(improved)

        if len(self._improvement_window) == 2 and not any(self._improvement_window):
            self._stagnation_count += 1
        elif improved:
            self._stagnation_count = 0

        self._prev_hv_for_stagnation = current_hv
        self._prev_pareto_size = current_pareto_size
```

**实现逻辑：**
- **Log10 变换**：对 time 和 aging 取 log
- **归一化**：相对于 y_min, y_max 缩放到 [0,1]
- **Tchebycheff**：`F_tch = max_i(w_i · Ȳ_i) + η · Σ(w_i · Ȳ_i)`
- **找最优**：`f_min = min(F_tch)`, `theta_best = argmin(F_tch)`
- **停滞检测**：滑动窗口检测 HV 或 PF 是否连续 2 次无改进

**方法：`compute_hypervolume()` (第439-461行)**

```python
def compute_hypervolume(self, ref_point: Optional[np.ndarray] = None) -> float:
    ref = ref_point if ref_point is not None else self.ref_point
    _, Y_pf = self.get_pareto_XY()

    if len(Y_pf) == 0:
        return 0.0

    # 1. Log10 变换（time 和 aging）
    Y_hv = Y_pf.copy()
    Y_hv[:, 0] = np.log10(np.maximum(Y_pf[:, 0], 1.0))
    Y_hv[:, 2] = np.log10(np.maximum(Y_pf[:, 2], 1e-12))

    # 2. 参考点也变换
    ref_hv = ref.copy()
    ref_hv[0] = np.log10(ref[0])
    ref_hv[2] = np.log10(ref[2])

    # 3. 过滤支配 ref 的点
    mask = np.all(Y_hv < ref_hv, axis=1)
    Y_hv = Y_hv[mask]

    if len(Y_hv) == 0:
        return 0.0

    # 4. 计算 3D 超体积
    hv_raw = self._hv_3d(Y_hv, ref_hv)

    # 5. 归一化
    return hv_raw / self.hv_max
```

**实现逻辑：**
- **变换**：对 time 和 aging 取 log10
- **过滤**：只保留支配 ref 的点（Y_hv < ref_hv）
- **3D HV 计算**：递归扫描线，转换为 2D HV 累加
- **归一化**：`HV / HV_max`（HV_max 为理想超体积）

```python
def evaluate(self, theta) -> Dict:
    rng = np.random.get_state()
    try:
        theta_arr = np.asarray(theta, dtype=float).ravel()
        if theta_arr.size < 5:
            return self._penalty("theta 必须包含 5 个决策变量")
```

**文件：`pybamm_simulator.py`**
**方法：`_run()` (第233-354行)**

```python
def _run(self, I1, I2, I3, dSOC1, dSOC2) -> Dict:
    dSOC3 = SOC_SPAN - dSOC1 - dSOC2  # SOC_SPAN = 0.8

    # 1. 电流换算
    if self.use_crate:
        I1_A = I1 * self.Q_eff  # C 倍率模式
        I2_A = I2 * self.Q_eff
        I3_A = I3 * self.Q_eff
    else:
        I1_A = I1 * self.Q_eff / 5.0  # 默认模式（协议参数）
        I2_A = I2 * self.Q_eff / 5.0
        I3_A = I3 * self.Q_eff / 5.0

    # 2. 计算各段时间
    if self.use_crate:
        t1 = self.Q_eff / I1_A * dSOC1 * 3600.0
        t2 = self.Q_eff / I2_A * dSOC2 * 3600.0
        t3 = self.Q_eff / I3_A * dSOC3 * 3600.0
    else:
        t1 = 5.0 * self.SOH / I1 * dSOC1 * 3600.0
        t2 = 5.0 * self.SOH / I2 * dSOC2 * 3600.0
        t3 = 5.0 * self.SOH / I3 * dSOC3 * 3600.0

    # 3. 构建 SPMe 模型
    if self.aging_mode in ("physical", "both"):
        model = pybamm.lithium_ion.SPMe(options={
            "thermal": "lumped",
            "SEI": "reaction limited",
            "lithium plating": "irreversible",
        })
    else:
        model = pybamm.lithium_ion.SPMe(options={"thermal": "lumped"})

    # 4. 设置参数
    param = pybamm.ParameterValues("Chen2020")
    param.update({"Current function [A]": "[input]"})
    param.update(_IDENTIFIED_PARAMS, check_already_exists=True)

    if self.aging_mode in ("physical", "both"):
        param.update(_PLATING_PARAMS, check_already_exists=False)

    param.update({
        "Nominal cell capacity [A.h]": self.Q_eff,
        "Upper voltage cut-off [V]": self.V_max + 0.1,
        "Initial temperature [K]": self.T_init,
        "Ambient temperature [K]": self.T_init,
    })

    # 5. 设置初始化化学计量比
    try:
        param.set_initial_stoichiometries(f"{self.V_init} V")
    except Exception:
        logger.warning("set_initial_stoichiometries 失败，回退默认初值")

    # 6. 分段求解（3 个 CC 阶段）
    voltage_all = [self.V_init]
    temp_all = [self.T_init]
    soc_all = []
    current_all = [0.0]

    c0 = float(param["Initial concentration in negative electrode [mol.m-3]"])
    soc_all.append(_neg_conc_to_soc(c0))

    last_sol = None
    stage_currents = [I1_A, I2_A, I3_A]
    stage_times = [t1, t2, t3]

    for stage_I, stage_t in zip(stage_currents, stage_times):
        if last_sol is not None:
            model.set_initial_conditions_from(last_sol)  # 连续求解

        sim = pybamm.Simulation(model, parameter_values=param)
        st = max(1, int(round(stage_t)))
        t_eval = np.linspace(0.0, float(st), st + 1)

        sol = sim.solve(t_eval, inputs={"Current function [A]": -stage_I})

        v_stage = np.asarray(sol["Voltage [V]"].entries, dtype=float).reshape(-1)
        t_stage = np.asarray(sol["X-averaged cell temperature [K]"].entries, dtype=float).reshape(-1)
        c_stage = np.asarray(sol["R-averaged negative particle concentration [mol.m-3]"].entries, dtype=float)
        soc_stage = self._soc_series_from_conc(c_stage, len(v_stage))

        # 拼接轨迹（去掉首点，避免重复）
        voltage_all.extend(v_stage[1:].tolist())
        temp_all.extend(t_stage[1:].tolist())
        soc_all.extend(soc_stage[1:].tolist())
        current_all.extend([float(stage_I)] * max(0, len(v_stage) - 1))

        last_sol = sol

    planned_total_time = float(sum(int(round(t)) for t in stage_times))

    return self._extract_from_series(
        voltage_all=np.asarray(voltage_all),
        temp_all=np.asarray(temp_all),
        soc_all=np.asarray(soc_all),
        current_all=np.asarray(current_all),
        last_sol=last_sol,
        total_timeTime_override=planned_total_time,
    )
```

**实现逻辑：**
- **dSOC3 计算**：`dSOC3 = 0.8 - dSOC1 - dSOC2`
- **电流换算**：协议参数 `I` → 仿真电流 `I_A = I · Q_eff / 5`
- **时间计算**：`t = (5 · SOH / I) · dSOC · 3600`
- **SPMe 模型**：Single Particle Model with Electrolyte + 热模型
- **参数设置**：Chen2020 参数集 + 辨识热参数
- **分段求解**：3 个 CC 阶段连续求解，拼接轨迹
- **约束检查**：T_peak ≤ T_max, V_peak ≤ V_max

```python
def _extract_from_series(self, voltage_all, temp_all, soc_all, current_all,
                        last_sol, total_time_override) -> Dict:
    total_time = float(total_time_override)
    peak_temp = = float(np.max(temp_all))
    peak_volt = float(np.max(voltage_all))
    soc_final = float(soc_all[-1])

    # 1. 约束检查
    if peak_temp > self.T_max:
        return self._penalty(f"峰值温度 {peak_temp:.2f} K > 上限 {self.T_max} K")
    if peak_volt > self.V_max:
        return self._penalty(f"峰值电压 {peak_volt:.3f} V > 上限 {self.V_max} V")

    # 2. 温升 ΔT
    delta_temp_raw = peak_temp - 298.15  # 相对 25°C
    delta_temp = TEMP_RISE_OBJ_ALIGN * delta_temp_raw  # 1.10 对齐系数

    # 3. 老化计算
    mean_soc_pct = float(np.mean(soc_all) * 100)
    mean_temp_raw = float(np.mean(temp_all))
    mean_temp = 298.15 + TEMP_FOR_AGING_ALIGN * (mean_temp_raw - 298.15)
    mean_current = float(np.mean(current_all))

    aging_empirical = self._aging_empirical(mean_soc_pct, mean_temp, mean_current)

    # 4. 物理老化（可选）
    aging_physical = None
    if self.aging_mode in ("physical", "both") and last_sol is not None:
        aging_physical = self._aging_physical(last_sol)

    # 5. 选择老化指标
    aging_obj = aging_physical if self.aging_mode == "physical" else aging_empirical

    # 6. 轨迹输出
    trajectories = {
        "V": voltage_all.tolist(),
        "T": temp_all.tolist(),
        "SOC": soc_all.tolist(),
        "I": current_all.tolist(),
        "time": np.arange(len(soc_all), dtype=float).tolist(),
    }

    return {
        "raw_objectives": np.array([total_time, delta_temp, aging_obj]),
        "soc_final": soc_final,
        "feasible": True,
        "violation": None,
        "trajectories": trajectories,
    }
```

**实现逻辑：**
- **约束检查**：T_peak ≤ 328.15 K (55°C), V_peak ≤ 4.4 V
- **温升计算**：`ΔT = 1.10 × (T_peak - 298.15)`
- **老化计算**：经验公式 `aging = Q_eff / cal_cap(...) × 100`
- **mean 特征**：计算平均 SOC、温度、电流
- **返回**：3D 目标 `[time_s, delta_temp_K, aging_%]`
