## 4. 集成新 LLM 后端

### 4.1 添加 Azure OpenAI 支持

```python
# llm_interface.py
class LLMCaller:
    def __init__(self, config: LLMConfig):
        self.config = config

        if config.backend == "azure_openai":
            from openai import AzureOpenAI
            self.client = AzureOpenAI(
                azure_endpoint=config.api_base,
                api_key=config.api_key,
                api_version="2024-02-15-preview",
            )
        # ... 其他后端 ...

    def call(self, prompt: str, system_prompt: str = None) -> str:
        if self.config.backend == "azure_openai":
            messages = []
            if system_prompt:
                messages.append({"role": "system", "content": system_prompt})
            messages.append({"role": "user", "content": prompt})

            response = self.client.chat.completions.create(
                model=self.config.model,
                messages=messages,
                temperature=self.config.temperature,
                max_tokens=self.config.max_tokens,
            )
            return response.choices[0].message.content
```

## 6. 性能优化

### 6.1 并行评估候选点

```python
# 原代码：顺序评估
for theta in selected_thetas:
    result = simulator.evaluate(theta)

# 并行版本
from concurrent.futures import ThreadPoolExecutor

def evaluate_batch(thetas):
    with ThreadPoolExecutor(max_workers=4) as executor:
        results = list(executor.map(simulator.evaluate, thetas))
    return results

results = evaluate_batch(selected_thetas)
```

### 6.2 缓存 PyBaMM 模型

```python
class CachedPyBaMMSimulator(PyBaMMSimulator):
    def __init__(self):
        super().__init__()
        self._model_cache = {}
        self._param_cache = {}

    def _run(self, I1, SOC1, I2):
        # 检查缓存
        key = (I1, SOC1, I2)
        if key in self._model_cache:
            return self._model_cache[key]

        # 运行仿真
        result = super()._run(I1, SOC1, I2)
        self._model_cache[key] = result
        return result
```
